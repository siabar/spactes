// date_r0 (Timestamp style, full date)
// EXAMPLE r0a: 29-01-2010,  29 01 2010, 29.01.2010,  29/01/2010, 29/XI/2010, 29-01/2010,...
// EXAMPLE r0b: 29-01-10,  29 01 10, 29.01.10,  29/01/10, 29/XI/10, 29-01/10,...
// EXAMPLE r0c: 29-Ago-2010,  29-Agosto-2010, 29 Ago 2010, 29.agosto/2010,...
// EXAMPLE r0d: 29-Ago-10,  29-Agosto-10, 29 Ago 10, 29.agosto/10,...
RULENAME="date_r0a",EXTRACTION="(%reDayNumber(/|\.|-)%reMonthNumber(/|\.|-)%reYear4Digit)",NORM_VALUE="group(6)-%normMonth(group(4))-%normDay(group(2))"
RULENAME="date_r0b",EXTRACTION="(%reDayNumber(/|\.|-)%reMonthNumber(/|\.|-)%reYear2Digit)",NORM_VALUE="%normYearBC(group(6))-%normMonth(group(4))-%normDay(group(2))"
RULENAME="date_r0c",EXTRACTION="(%reDayNumber(/| |\.|-)(%reMonthLong|%reMonthShort)(/| |\.|-)%reYear4Digit)",NORM_VALUE="group(8)-%normMonth(group(4))-%normDay(group(2))"
RULENAME="date_r0d",EXTRACTION="(%reDayNumber(/| |\.|-)(%reMonthLong|%reMonthShort)(/| |\.|-)%reYear2Digit)",NORM_VALUE="%normYearBC(group(8))-%normMonth(group(4))-%normDay(group(2))"


// date_r1 (full date)
// EXAMPLE r1a: el 30 de abril de 1999
// EXAMPLE r1a: el cinco de abril de 1999
// EXAMPLE r1b: 15 y 16 de abril de 1999 (find: 15 abril 1999)
// EXAMPLE r1c: enero 5 de 2012
RULENAME="date_r1a",EXTRACTION="(%rePartWords )?([Dd][Ee]l? |[Ee][Ll] |[Ll]a |[Aa]l )?(días? )?(%reDayNumber|%reDayWordTh|%reDayWord) ([Dd][Ee] )?(%reMonthLong|%reMonthShort) ([Dd]?[Ee][Ll]? )?%reYear4Digit(,? %reWeekday)?",NORM_VALUE="group(14)-%normMonth(group(10))-%normDay(group(5))",NORM_MOD="%normPartWords(group(2))",OFFSET="group(5)-group(14)"
RULENAME="date_r1b",EXTRACTION="(%rePartWords )?([Dd][Ee]l? |[Ee]l |[Ll]a |[Aa]l |[Ll][oa]s )?(días? )?(%reDayNumber|%reDayWordTh|%reDayWord)%reAndOrTo(%reDayNumber|%reDayWordTh|%reDayWord) de (%reMonthLong|%reMonthShort)( [Dd]?[Ee]l? )%reYear4Digit",NORM_VALUE="group(18)-%normMonth(group(14))-%normDay(group(5))",OFFSET="group(5)-group(5)",NORM_MOD="%normPartWords(group(2))"
RULENAME="date_r1c",EXTRACTION="(%rePartWords )?(%reMonthLong|%reMonthShort) (%reDayNumber|%reDayWordTh|%reDayWord)( [Dd]?[Ee]l? )%reYear4Digit",NORM_VALUE="group(11)-%normMonth(group(3))-%normDay(group(6))",NORM_MOD="%normPartWords(group(2))"


// date_r2 (day and month)
// EXAMPLE r2a: Este 06/11, día 03-10,...
// EXAMPLE r2b: El 06-Ago, Este 06/Septiembre, día 03-ene,...
// EXAMPLE r2c: el 5 de abril, del 5 de ene,...
// EXAMPLE r2c: el primero de abril
// EXAMPLE r2d: el 5 o 6 de abril (find: abril 5)

RULENAME="date_r2a",EXTRACTION="([Dd]el |[Ee]l |[Ee]ste |[Dd]ía |[Dd]ia )%reDayNumber(/| |\.|-)%reMonthNumber",NORM_VALUE="UNDEF-year-%normMonth(group(4))-%normDay(group(2))",OFFSET="group(2)-group(4)"
RULENAME="date_r2b",EXTRACTION="([Dd]el |[Ee]l |[Ee]ste |[Dd]ía |[Dd]ia )%reDayNumber(/| |\.|-)(%reMonthLong|%reMonthShort)",NORM_VALUE="UNDEF-year-%normMonth(group(4))-%normDay(group(2))",OFFSET="group(2)-group(4)"
RULENAME="date_r2c",EXTRACTION="(%rePartWords )?([Dd][Ee][Ll]? |[Ee][Ll] |[Ll]a |[Aa]l )?(días? )?(%reDayNumber|%reDayWordTh|%reDayWord) ([Dd][Ee] )(%reMonthLong|%reMonthShort)",NORM_VALUE="UNDEF-year-%normMonth(group(10))-%normDay(group(5))",NORM_MOD="%normPartWords(group(1))",OFFSET="group(5)-group(10)" 
RULENAME="date_r2d",EXTRACTION="(%reDayNumber|%reDayWordTh|%reDayWord)%reAndOrTo(%reDayNumber|%reDayWordTh|%reDayWord) de (%reMonthLong|%reMonthShort)",NORM_VALUE="%normMonth(group(10))-%normDay(group(1))",OFFSET="group(1)-group(1)"
RULENAME="date_r2e",EXTRACTION="([Dd][Ee]l? |[Ee]l |[Ll]a |[Aa]l |[Ll][oa]s )(días? )?(%reDayNumber|%reDayWordTh|%reDayWord)%reAndOrTo([Dd][Ee]l? |[Ee]l |[Ll]a |[Aa]l |[Ll][oa]s )(días? )?(%reDayNumber|%reDayWordTh|%reDayWord) de (%reMonthLong|%reMonthShort)",NORM_VALUE="UNDEF-year-%normMonth(group(14))-%normDay(group(3))",OFFSET="group(1)-group(3)"
RULENAME="date_r2f",EXTRACTION="(%reDayNumber|%reDayWordTh|%reDayWord)(/|\.|-)(%reMonthNumber|%reMonthLong|%reMonthShort)[;|:|,]?[ ]?[0-2]?[0-9][:|.][0-5][0-9][h|horas]?",NORM_VALUE="UNDEF-year-%normMonth(group(6))-%normDay(group(1))",OFFSET="group(1)-group(6)"
RULENAME="date_r2g",EXTRACTION="\((%reDayNumber|%reDayWordTh|%reDayWord)(/|\.|-)(%reMonthNumber|%reMonthLong|%reMonthShort)[\)|;|:|,]",NORM_VALUE="UNDEF-year-%normMonth(group(6))-%normDay(group(1))",OFFSET="group(1)-group(6)"
RULENAME="date_r2h",EXTRACTION="(%reDayNumber|%reDayWordTh|%reDayWord)/(%reMonthNumber|%reMonthLong|%reMonthShort)[\)|:|]",NORM_VALUE="UNDEF-year-%normMonth(group(5))-%normDay(group(1))",OFFSET="group(1)-group(5)"
